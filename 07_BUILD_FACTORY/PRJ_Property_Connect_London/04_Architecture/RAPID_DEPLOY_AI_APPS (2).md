```python
# prospect_finder.py

import requests
from typing import List, Dict

class PropertyProspectFinder:
    def __init__(self, google_api_key: str):
        self.api_key = google_api_key
        self.prospects = []
    
    def find_estate_agents(self, location: str = "London") -> List[Dict]:
        """Find estate agents using Google Places API"""
        
        # Search queries
        queries = [
            f"estate agents {location}",
            f"letting agents {location}",
            f"property management {location}",
        ]
        
        for query in queries:
            url = "https://maps.googleapis.com/maps/api/place/textsearch/json"
            params = {
                'query': query,
                'key': self.api_key
            }
            
            response = requests.get(url, params=params)
            places = response.json().get('results', [])
            
            for place in places:
                # Get website from place details
                details = self.get_place_details(place['place_id'])
                
                self.prospects.append({
                    'name': place['name'],
                    'address': place.get('formatted_address'),
                    'rating': place.get('rating'),
                    'reviews': place.get('user_ratings_total'),
                    'website': details.get('website'),
                    'phone': details.get('formatted_phone_number'),
                    'place_id': place['place_id'],
                    'category': 'estate_agent'
                })
        
        return self.prospects
    
    def find_home_furnishing(self, location: str = "London") -> List[Dict]:
        """Find home furnishing businesses"""
        
        queries = [
            f"furniture stores {location}",
            f"interior design showroom {location}",
            f"kitchen showroom {location}",
            f"bathroom showroom {location}",
            f"home decor {location}",
            f"sofa shop {location}",
            f"bed shop {location}",
        ]
        
        for query in queries:
            # Same API call pattern
            results = self.search_places(query)
            for place in results:
                place['category'] = 'home_furnishing'
                self.prospects.append(place)
        
        return self.prospects
    
    def get_place_details(self, place_id: str) -> Dict:
        """Get detailed info including website"""
        url = "https://maps.googleapis.com/maps/api/place/details/json"
        params = {
            'place_id': place_id,
            'fields': 'website,formatted_phone_number,opening_hours',
            'key': self.api_key
        }
        response = requests.get(url, params=params)
        return response.json().get('result', {})
    
    def export_csv(self, filename: str = "prospects.csv"):
        """Export for outreach"""
        import csv
        
        with open(filename, 'w', newline='') as f:
            writer = csv.DictWriter(f, fieldnames=[
                'name', 'category', 'website', 'phone', 
                'rating', 'reviews', 'address'
            ])
            writer.writeheader()
            writer.writerows(self.prospects)


# Usage

    "Lambeth", "Wandsworth", "Hammersmith", "Fulham",

]

for borough in boroughs:
    finder.find_estate_agents(f"{borough}, London")
    finder.find_home_furnishing(f"{borough}, London")

finder.export_csv("london_property_prospects.csv")
print(f"Found {len(finder.prospects)} prospects")
```

```python
        )
        
        return json.loads(analysis.content[0].text)
    
    def extract_branding(self, url: str) -> Dict:
        """Extract brand elements for white-labeling"""
        
        # Use Firecrawl for structured extraction
        response = requests.post(
            "https://api.firecrawl.dev/v0/scrape",
            headers={"Authorization": f"Bearer {FIRECRAWL_KEY}"},
            json={
                "url": url,
                "extractorOptions": {
                    "mode": "llm-extraction",
                    "extractionPrompt": """
Extract brand elements:
{
  "company_name": "exact company name",
  "primary_color": "#hex",
  "secondary_color": "#hex", 
  "logo_url": "URL to logo image",
  "tagline": "if present",
  "font_family": "main font used",
  "brand_tone": "professional/friendly/luxury/etc"
}
"""
                }
            }
        )
        
        return response.json()['data']['llm_extraction']
    
    def capture_screenshot(self, url: str) -> str:
        """Capture full page screenshot"""
        with sync_playwright() as p:
            browser = p.chromium.launch()
            page = browser.new_page(viewport={'width': 1920, 'height': 1080})
            page.goto(url, wait_until='networkidle')
            screenshot = page.screenshot(full_page=True)
            browser.close()
            return base64.b64encode(screenshot).decode()
    
    def batch_evaluate(self, urls: List[str]) -> List[Dict]:
        """Evaluate multiple sites"""
        results = []
        for url in urls:
            try:
                eval_result = self.evaluate_site(url)
                brand = self.extract_branding(url)
                results.append({
                    'url': url,
                    'evaluation': eval_result,
                    'branding': brand
                })
            except Exception as e:
                results.append({'url': url, 'error': str(e)})
        return results
```

```python
# Create outreach priority list

def prioritize_prospects(evaluated_prospects: List[Dict]) -> List[Dict]:
    """Score and rank prospects for outreach"""
    
    for prospect in evaluated_prospects:
        eval_data = prospect.get('evaluation', {})
        
        # Calculate priority score
        score = 0
        
        # High value gaps
        if not eval_data.get('has_chatbot'):
            score += 30
        if not eval_data.get('has_online_booking'):
            score += 25
        if not eval_data.get('has_calculators'):
            score += 15
        if not eval_data.get('has_lead_capture'):
            score += 20
        
        # Business signals
        if eval_data.get('estimated_tech_budget') == 'high':
            score += 20
        elif eval_data.get('estimated_tech_budget') == 'medium':
            score += 10
        
        if eval_data.get('design_quality', 0) >= 7:
            score += 10  # Professional site = likely to invest
        
        prospect['priority_score'] = score
    
    # Sort by priority
    return sorted(evaluated_prospects, key=lambda x: x['priority_score'], reverse=True)
```
