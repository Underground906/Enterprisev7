-- Category taxonomy
categories (
  id, 
  pillar ENUM('buying','selling','renting','letting','investing','renovating','building'),

  name,
  parent_id,

)

professional_specializations (
  professional_id,
  category_id,

  years_experience INTEGER,

user_journeys (
  id,
  user_id,
  pillar,
  journey_type VARCHAR, -- 'first_time_buyer', 'downsizer', etc.
  current_stage VARCHAR,
  started_at,
  completed_at,
  milestones JSONB
)

-- AI recommendations log
ai_recommendations (
  id,
  user_id,
  recommendation_type, -- 'property', 'professional', 'content', 'area'
  recommendation_data JSONB,
  shown_at,
  clicked BOOLEAN,
  converted BOOLEAN
)

-- Verification records
verifications (
  id,
  entity_type ENUM('user','professional'),
  entity_id,
  verification_type, -- 'email', 'phone', 'id', 'credential', 'insurance'
  status ENUM('pending','verified','rejected','expired'),
  verified_at,
  expires_at,
  provider VARCHAR, -- 'onfido', 'companies_house', 'manual'
  evidence_url
)

-- Sustainability data
property_sustainability (
  property_id,
  epc_rating CHAR(1),
  epc_score INTEGER,
  annual_energy_cost DECIMAL,
  carbon_footprint DECIMAL,
  green_features TEXT[],
  renewable_energy BOOLEAN,
  insulation_rating VARCHAR
)

-- Events
events (
  id,
  host_id,
  host_type ENUM('professional','platform'),
  event_type, -- 'webinar', 'viewing', 'open_house', 'networking'
  title,
  description,
  start_time,
  end_time,
  location_type ENUM('online','in_person','hybrid'),
  location_details JSONB,
  max_attendees INTEGER,
  registration_required BOOLEAN,
  recording_url
)

event_attendees (
  event_id,
  user_id,
  status ENUM('registered','attended','no_show','cancelled'),
  registered_at
)
